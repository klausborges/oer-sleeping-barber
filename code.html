void *barber() {
  while (1) {
    /* Barber goes to sleep if there are no customers */
    sem_wait(&customers);

    /* Holds the mutex */
    sem_wait(&mutex);

    /* Barber brings one customer to the barber chair */
    waiting_customers -= 1;

    /* Barber is now ready to work */
    sem_post(&barbers);

    /* Releases the mutex */
    sem_post(&mutex);

    cut_hair();
  }

  pthread_exit(NULL);
}

void *customer() {
  /* Holds the mutex */
  sem_wait(&mutex);

  /* Customer waits if there's an empty waiting chair */
  if (waiting_customers < N_CHAIRS) {      
    /* Customer waits */
    waiting_customers++;

    /* Barber gets notified of the arrival */  
    sem_post(&customers);

    /* Releases the mutex */
    sem_post(&mutex);

    /* Waits to be called by the barber */
    sem_wait(&barbers);

    get_haircut();
  }
  /* Customer leaves if there are no empty waiting chairs */
  else {
    /* Releases the mutex */
    sem_post(&mutex);
    leave_barbershop();
  }
}
